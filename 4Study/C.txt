★ ++

int main(){
     int a =10;
     int b =20;
     printf("%d\n",a); // a = 10
     a ++
     printf("%d\n",a); // a = 11

     printf("%d\n",++b); // b = 21 // 先に足す
     printf("%d\n",b++); // b = 21 // 後で足す
     printf("%d\n",b);  // b = 22
}

☆ポインタ引数
// &引数でアドレスに変換
// *アドレスで文字または数値に変換

int main(){
     int a = 1;
     int pointer = &a; // ポインタ引数
     printf("pointer adress : %d\t pointer : %d",pointer,*pointer);

     int b = 2;
     int pointer = &b;
     *pointer = *pointer *3;
     printf("pointer adress : %d\t pointer *3 : %d,pointer,*pointer);
}
