★ regular expression [group]

import re

pattern = re.compile("ca.e")
.  : 하나의 문자      ex) (ca.e) > care, cafe... | caffe (x)
^ : 문자열의 시작    ex) (^de) > desk, destination... | fade(x)
$ : 문자열 끝         ex) (se$)  > case, base... | face (x)

match=pattern.match("cafe") #match : 주어진 문자열의 처분부터 일치하는지 확인

def print_match(m):
     if match:        >     (== if match in match.group())
          print(match.group())
     else:
          print("not matched")

print_match(match)
