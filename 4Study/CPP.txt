★ Class

☆クラス定義
class MikanBox{
public:
     MikanBox();
     ~MikanBox();
     void Add(int addmikan);
     void Del(int delmikan);
     void Empty();
     int Gettotal(){
          return total;
     }
private:
     int total;
/* 同じオブジェクトのメンバ関数からのみ、アクセスできる*/
};

void MikanBox::Add(int addmikan){
     total += addmikan;
}

void MikanBox::Del(int delmikan){
     total -= delmikan;
     if (total < 0){
          total=0;
     }
}

void MikanBox::Empty(){
     total=0;
}

☆コンストラクタ
MikanBox::MikanBox(){
     total=0;
}

☆デストラクタ
MikanBox::~MikanBox(){
     std::cout << "みかん箱オブジェクトの任務終了\n"
}

int main(){
     MikanBox mymikanBox;
     /*オブジェクト作成*/
     myMikanBox.Empty();
     myMikanBox.Add(5);
     myMikanBox.Del(2);
     std::cout << "箱の中のみかん :" << myMikanBox.Gettotal << "個\n";
     return 0;
}
